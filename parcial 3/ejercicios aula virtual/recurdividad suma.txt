// Declaraciï¿½n de la funciï¿½n recursiva
int sumaRecursiva(int n) {
	if (n == 0) {
		return 0; // Caso base
	} else {
		return n + sumaRecursiva(n - 1); // Caso recursivo
	}
}

int main() {
	int numero; // Cambie este valor para probar con otros nï¿½meros
	int resultado;
	printf("Ingresa el nï¿½mero de tï¿½rminos a mostrar: ");
	scanf("%d", &numero);
	// Validaciï¿½n para evitar nï¿½meros negativos
	if (numero < 0) {
		printf("Por favor, ingrese un nï¿½mero entero positivo.\n");
		return 1;
	}
	
	resultado = sumaRecursiva(numero);
	
	printf("La suma de los primeros %d numeros naturales es: %d\n", numero, resultado);
	
	return 0;
}
